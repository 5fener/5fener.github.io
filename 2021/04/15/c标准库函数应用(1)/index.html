  • #include<math.h>

    1. abs函数
      函数定义：int abs（int i）；
      函数功能：求整数的绝对值

    2. labs函数
      函数定义：long labs(long n);
      函数功能：求长整数的绝对值

    3. fabs函数
      函数定义：double fabs(double x);
      函数功能：返回浮点数的绝对值

    4. sin函数
      函数定义：double sin(double x);
      函数功能：求解正弦

    5. cos函数
      函数定义：double cos(double x);
      函数功能：求解余弦

    6. tan函数
      函数定义: double tan(double x);
      函数功能：求解正切

    以下举例给出1~3的程序代码：

‘’’
#include<stdio.h>
#include<math.h>

int main()
{
    int iAbsoluteNumber;
    int iNumber1 = -12;
    iAbsoluteNumber = abs(iNumber1);
    printf("%d\n", iAbsoluteNumber);

    long iResult;
    long iNumber2 = -1234567890L;
    iResult = labs(iNumber2);
    printf("%ld\n", iResult);

    double fFloatResult;
    double fNumber = -1234.0;
    fFloatResult = fabs(fNumber);
    printf("%lf", fNumber);

    return 0;
}

  • #include<ctype.h>

    1. isalpha函数
      函数定义：int isalpha(int ch);
      函数功能：检测字符是否为字母

    2. isdigit函数
      函数定义：int isdigit(int ch);
      函数功能：检测字符是否为数字

    3. isalnum函数
      函数定义：int isalnum(int ch);
      函数功能：检测字符是否为字母或数字

    以下举例给出4~6的程序代码：

‘’’
#include<stdio.h>
#include<ctype.h>

void SwitchShow(char c);

int main()
{
    char cCharPut;
    char cCharTemp;

    printf("First enter:");
    scanf_s("%c", &cCharPut,sizeof(cCharPut));
    SwitchShow(cCharPut);
    cCharTemp = getchar();

    printf("Second enter:");
    scanf_s("%c", &cCharPut,sizeof(cCharPut));
    SwitchShow(cCharPut);
    cCharTemp = getchar();

    printf("Third enter:");
    scanf_s("%c", &cCharPut,sizeof(cCharPut));
    SwitchShow(cCharPut);

    return 0;
}

void SwitchShow(char cChar)
{
    if (isalpha(cChar))
        printf("You entered a letter of  the alphabet %c\n", cChar);
    if (isdigit(cChar))
        printf("You entered the digit %c\n", cChar);
    if (isalnum(cChar))
        printf("You entered the alphanumuric character %c\n", cChar);
    else
        printf("You entered the character is not alphabet or digit:%c\n", cChar);
}
